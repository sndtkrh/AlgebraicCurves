\section{Affine Space and Algebraic Sets}
\begin{prob}
  $\A^{1}(k)$の代数的部分集合は有限部分集合であることを示せ．
\end{prob}
\begin{ans}
  $X \subset \A^1(k)$を代数的部分集合であるとする．
  $X = V( S )$となる$k[X]$の部分集合$S \ne \emptyset$が存在する．
  $f \in S$を取る．
  $f = \sum_{i=0}^n a_i X^i, a_i \in k$
  と書ける．
  $f$の根は高々$n$個しかない．
  よって$X = V(S) \subseteq V(f)$であるから
  $X$は有限集合である．
\end{ans}
